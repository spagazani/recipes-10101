ingredients:
  - section: "ALL"
    items:
      - "16 ounces ketchup"
      - "8 ounces water"
      - "3 ounces brown sugar"
      - "3 ounces white sugar"
      - "1/2 Tbsp pepper"
      - "1/2 Tbsp onion powder"
      - "1/2 Tbsp dry mustard powder"
      - "1-ounce lemon juice"
      - "1-ounce Worcestershire sauce"
      - "4 ounces apple cider vinegar"
      - "1-ounce light corn syrup"
      - "1 Tbsp paprika"
      - "1/2 tsp onion powder"
instructions:
  - "Combine all ingredients in a stockpot or large Dutch oven. Bring to a boil over high heat stirring frequently to prevent sticking"
  - "Reduce temperature to low and simmer, uncovered, for at least 2 hours."
