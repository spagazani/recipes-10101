title: "Gramma's Baked Beans"
ingredients:
  - section: "ALL"
    items:
      - "2 lbs navy beans"
      - "2 onions"
      - "1 Tbsp salt"
      - "4 tsp cider vinegar"
      - "1 tsp dry mustard"
      - "2 Tbsp brown sugar"
      - "1/2 cup molasses"
      - "1 cup ketchup"
      - "Pepper"
      - "Water as needed"
      - "1/2 lb salt pork"
instructions:
  - "Soak beans overnight then simmer for 30 minutes."
  - "Combine all ingredients in a bean or large casserole dish. Bake at 250F for 7 hours."
