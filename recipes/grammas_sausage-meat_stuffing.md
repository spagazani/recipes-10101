title: "Gramma's Sausage-Meat Stuffing"
ingredients:
  - section: "ALL"
    items:
      - "1 cup onions"
      - "2 Tbsp butter"
      - "1 cup celery"
      - "1 tsp poultry seasoning"
      - "1 lb sausage-meat"
      - "4 cups fresh breadcrumbs"
      - "Salt and pepper"
instructions:
  - "Sauté onions and celery in butter and then stir in sausage-meat."
  - "Mix with ingredients and stuff turkey. Stuffing can be refrigerated over night."
