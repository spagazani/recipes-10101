tags: unclassified
title: "Mom's Spaghetti Sauce"
ingredients:
  - section: "ALL"
    items:
      - "1/2 cup parmesan cheese"
      - "4 Tbsp margarine"
      - "1 1/2 cups onion"
      - "2 cloves garlic"
      - "3 lbs lean ground beef"
      - "1 green pepper"
      - "2 28oz cans of tomatoes"
      - "2 7 1/2 oz cans tomato sauce"
      - "2 5 1/2 oz cans tomato paste"
      - "1 Tbsp sugar"
      - "1 Tbsp salt"
      - "2 tsp oregano"
      - "1 1/2 tsp basil"
      - "1/2 tsp chili powder"
      - "1/4 tsp pepper"
instructions:
  - "Cook margarine, onion, and garlic in large saucepan for 8 minutes or until onion is tender."
  - "Add ground beef and continue cooking for an additional 15 minutes or until beef is browned. Add remaining ingredients and simmer for 1 hour."
