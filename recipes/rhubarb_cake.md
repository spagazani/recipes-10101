title: "Rhubarb Cake"
ingredients:
  - section: "ALL"
    items:
      - "1 1/2 cup brown sugar"
      - "1/2 cup margarine"
      - "1 beaten egg"
      - "1 tsp vanilla"
      - "2 cups flour"
      - "1 cup sour milk"
      - "1 tsp baking soda"
      - "2 cups cut rhubarb"
      - "1/4 cup sugar"
      - "1 tsp cinnamon"
instructions:
  - "Mix first 7 ingredients. Add raw rhubarb. Place in a 13x9” greased pan."
  - "Mix together sugar and cinnamon. Sprinkle on top. Bake at 350F for 40 minutes."
