title: "Michelle's Zucchini Bread"
ingredients:
  - section: "ALL"
    items:
      - "1 cup egg substitute"
      - "1 tsp baking powder"
      - "1 1/2 cups sugar"
      - "1 tsp baking soda"
      - "1 Tbsp vanilla"
      - "2 tsp cinnamon"
      - "1/2 cup oil"
      - "1/2 tsp cocoa"
      - "2 cups zucchini, ground"
      - "1/2 tsp cloves"
      - "3 cups flour"
      - "1/4 tsp allspice"
instructions:
  - "In a large bowl, mix first 5 ingredients, egg substitute through zucchini. Set aside."
  - "Combine dry ingredients in a separate bowl."
  - "Add dry ingredients to zucchini mixture and pour into two 9x5 loaf pans."
  - "Bake at 350F for 45 to 60 minutes. Makes 12 slices per loaf."
